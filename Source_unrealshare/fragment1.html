simulated function CalcVelocity(vector Momentum, float ExplosionSize)
{
    ExplosionSize = VSize(Momentum);
    Velocity = VRand()*(ExplosionSize+FRand()*100.0+100.0); 
    Velocity.z += ExplosionSize/2;
}
