function ForceMeshToExist()
{
    //never called
    Spawn(class 'Nali');
}

static simulated function bool AllowChunk(int N, name A)
{
    if ( (A == 'Dead3') && (N == 6) )
        return false;

    return true;
}

function CreateReplacement()
{
    local CreatureChunks carc;
    
    if (bHidden)
        return;
    carc = Spawn(class'NaliMasterChunk'); 
    if (carc != None)
    {
        carc.bMasterChunk = true;
        carc.Initfor(self);
        carc.Bugs = Bugs;
        if ( Bugs != None )
            Bugs.SetBase(carc);
        Bugs = None;
    }
    else if ( Bugs != None )
        Bugs.Destroy();
}
