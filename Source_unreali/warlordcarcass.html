function AnimEnd()
{
    if ( AnimSequence == 'Dead2A' )
    {
        if ( Physics == PHYS_None )
        {
            LieStill();
            PlayAnim('Dead2B', 0.7, 0.07);
        }
        else
            LoopAnim('Fall');
    } 
    else if ( Physics == PHYS_None )
        LieStill();
}

function Landed(vector HitNormal)
{
    if ( AnimSequence == 'Fall' )
    {
        LieStill();
        PlayAnim('Dead2B', 0.7, 0.07);
    }
    SetPhysics(PHYS_None);
}

    }
