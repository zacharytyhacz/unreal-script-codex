function ForceMeshToExist()
{
    //never called
    Spawn(class 'Krall');
}

static simulated function bool AllowChunk(int N, name A)
{
    if ( (A == 'Dead5') && (N == 4) )
        return false;
    if ( (A == 'LeglessDeath') && (N == 2) )
        return false;

    return true;
}

function InitFor(actor Other)
{
    Super.InitFor(Other);
    if ( AnimSequence == 'LeglessDeath' )
        SetCollision(true, false, false);
}
