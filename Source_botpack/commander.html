{
    Velocity = vect(0,0,0);
    Acceleration = vect(0,0,0);
    BaseEyeHeight = Default.BaseEyeHeight;

    GotoState(PlayerReStartState);

// This pawn was possessed by a player.
function Possess()
{
    bIsPlayer = true;
    EyeHeight = BaseEyeHeight;
    NetPriority = 2;
    Weapon = None;
    Inventory = None;
    bCollideWorld = true;
    ClientReStart();    
}
